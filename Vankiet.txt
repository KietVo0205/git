import java.io.IOException;

	public static void main(String[] args) throws IOException {

	public boolean detele(String Path){
		File inputf=new File(Path);
		if(!inputf.exists())
		{
			System.out.println("no find file");
			return false;
		}
		else{
			if(inputf.isFile()){
				System.out.println("this is File");
			return false;
			}
			else{
				if(inputf.isDirectory()){
					File[] list=inputf.listFiles();
					for(File file:list){
						file.delete();
					}
					System.out.println("detele all file");
					System.out.println(inputf.getAbsolutePath());
					return true;
				}
			}
		}
		return false;
			
		}
	public boolean findFirst(String path,String paterrn){
		File inputf=new File(path);
		if(!inputf.exists())
		{
			System.out.println("no find file");
			return false;
		}
		else{
			if(inputf.isFile()){
				System.out.println("this is File");
			return false;
			}
			else{
				if(inputf.isDirectory()){
					File[] list=inputf.listFiles();
					for(File file:list){
						if(file.isFile()&&file.getPath().equals(paterrn)){
			

							return true;
							}
						else{
							if(file.isDirectory()){
								boolean res=findFirst(file.getPath(), paterrn);
								if(res==true){
									return true;
								}
							}
						}
					}
		
	}
			}
			}
		return false;
		}
	}
